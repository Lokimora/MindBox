2) SELECT ProductId, COUNT(ProductId) FROM SALES s1
INNER JOIN
(
	SELECT CustomerId, MIN(DateCreated) as DateCreated FROM SALES
	GROUP BY CustomerId
)s2
on s1.CustomerId = s2.CustomerID AND s1.DateCreated = s2.DateCreated
GROUP BY ProductId
